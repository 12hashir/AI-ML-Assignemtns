    "import joblib\n",
    "import pandas as pd\n",
    "from sklearn.preprocessing import StandardScaler, LabelEncoder\n",
    "from sklearn.ensemble import RandomForestClassifier\n",
    "\n",
    "print('This cell simulates saving files from Task 9')\n",
    "\n",
    "# Load your dataset from Task 5\n",
    "df = pd.read_csv('Client_projects.csv')\n",
    "X = df.drop(['status', 'project_id', 'client_name', 'deadline'], axis=1)\n",
    "y = df['status']\n",
    "\n",
    "# Train + Save  --  Replace with your actual Task 9 trained model\n",
    "scaler = StandardScaler().fit(X)\n",
    "le     = LabelEncoder().fit(y)\n",
    "model  = RandomForestClassifier().fit(scaler.transform(X), le.transform(y))\n",
    "\n",
    "joblib.dump(model,  'optimized_model.pkl')\n",
    "joblib.dump(scaler, 'scaler.pkl')\n",
    "joblib.dump(le,     'label_encoder.pkl')\n",
    "\n",
    "print('Files saved: optimized_model.pkl, scaler.pkl, label_encoder.pkl')\n",
    "print('Note: Replace above code with your actual Task 9 trained model')\n"

    "app_code = 'import streamlit as st\\nimport joblib\\nimport numpy as np\\nimport pandas as pd\\n\\n# 1. PAGE SETUP + STYLING\\nst.set_page_config(page_title=\\'Project Status Predictor\\', page_icon=\\'🚀\\', layout=\\'centered\\')\\nst.title(\\'🚀 Project Status Prediction App\\')\\nst.write(\\'CoreTech Innovations - Task 10: AI Model Deployment\\')\\nst.markdown(\\'---\\')\\n\\n# 2. LOAD TRAINED MODEL + SCALER + LABEL ENCODER\\ntry:\\n    model         = joblib.load(\\'optimized_model.pkl\\')\\n    scaler        = joblib.load(\\'scaler.pkl\\')\\n    label_encoder = joblib.load(\\'label_encoder.pkl\\')\\nexcept Exception:\\n    st.error(\\'ERROR: Model files not found. Save them from Task 9 first.\\')\\n    st.stop()\\n\\n# 3. INPUT FORM\\nst.subheader(\\'Enter Project Details\\')\\ncol1, col2 = st.columns(2)\\n\\nwith col1:\\n    budget        = st.number_input(\\'Budget (PKR)\\',    min_value=1000, max_value=10000000, value=50000)\\n    team_size     = st.number_input(\\'Team Size\\',       min_value=1,    max_value=50,       value=3)\\n\\nwith col2:\\n    duration_days = st.number_input(\\'Duration (Days)\\', min_value=1,    max_value=365,      value=30)\\n    complexity    = st.selectbox(\\'Complexity\\', [\\'Low\\', \\'Medium\\', \\'High\\'])\\n\\n# 4. PREDICTION BUTTON\\nif st.button(\\'Predict Status\\', type=\\'primary\\'):\\n    complexity_map = {\\'Low\\': 0, \\'Medium\\': 1, \\'High\\': 2}\\n    input_data = np.array([[budget, team_size, duration_days, complexity_map[complexity]]])\\n\\n    # 5. SCALE + PREDICT\\n    input_scaled     = scaler.transform(input_data)\\n    prediction       = model.predict(input_scaled)\\n    predicted_status = label_encoder.inverse_transform(prediction)[0]\\n\\n    # 6. OUTPUT DISPLAY\\n    st.markdown(\\'---\\')\\n    st.subheader(\\'Prediction Result\\')\\n\\n    if predicted_status == \\'Completed\\':\\n        st.success(f\\'Predicted Status: **{predicted_status}**\\')\\n    elif predicted_status == \\'Delayed\\':\\n        st.warning(f\\'Predicted Status: **{predicted_status}**\\')\\n    else:\\n        st.error(f\\'Predicted Status: **{predicted_status}**\\')\\n\\n# 7. MODEL EXPLANATION\\nst.markdown(\\'---\\')\\nwith st.expander(\\'Model Explanation\\'):\\n    st.write(\"\"\"\\n    **Model Used**: RandomForestClassifier optimized with RandomizedSearchCV from Task 9\\n\\n    **How it works**:\\n    Takes project features (budget, team size, duration, complexity) and\\n    predicts whether the project will be Completed, Delayed, or Failed.\\n\\n    **Trained on**: Client_projects.csv from Task 5\\n    \"\"\")\\n\\nst.markdown(\\'---\\')\\nst.caption(\\'Submitted for CoreTech Innovations Internship - Task 10\\')\\n'\n",
